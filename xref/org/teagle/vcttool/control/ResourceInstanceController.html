126 	public void onEditConfig(final VctView vctView) {
127 		final ConfigurationDialog dialog = new ConfigurationDialog(
128 				vctView.getShell(), this.resourceInstance.getCommonName());
129 
130 		final List<? extends Configuration> configurations = this.resourceInstance
131 				.getConfigurations();
132 		for (final Configuration config : configurations) {
133 			final String name = config.getConfigParamAtomic().getCommonName();
134 			final String type = config.getConfigParamAtomic().getType();
135 			final String description = config.getConfigParamAtomic()
136 					.getDescription();
137 
138 			String defaultValue = config.getValue();
139 			if ((defaultValue == null) || defaultValue.isEmpty())
140 				defaultValue = config.getConfigParamAtomic().getDefaultValue();
141 			if (type.equals("boolean"))
142 				dialog.addConfigurationParameter(name,
143 						Boolean.parseBoolean(defaultValue), description);
144 			else if (type.equals("string") || type.startsWith("int")
145 					|| type.equals("float"))
146 				dialog.addConfigurationParameter(name, defaultValue,
147 						description);
148 		}
149 
150 		switch (dialog.show()) {
151 		case SWT.OK:
152 			for (final Configuration config : this.resourceInstance
153 					.getConfigurations()) {
154 				final String name = config.getConfigParamAtomic()
155 						.getCommonName();
156 				final String value = dialog.getConfiguratonParameter(name);
157 				if (value != null)
158 					config.setValue(value);
159 			}
160 
161 			break;
162 		case SWT.CANCEL:
163 			break;
164 		default:
165 			break;
166 		}

169 	public void onConnectionNew(final ConnectionEvent event) {
170 		final ResourceInstanceController targetControler = (ResourceInstanceController) event.target
171 				.getData();
172 
173 		final Object data = event.sourceLabel.getData();
174 		if (data == null) {
175 			for (final VctController vctControler : this.views.keySet())
176 				if (this.views.get(vctControler) == event.source) {
177 					targetControler.getResourceInstance().setParentInstance(
178 							this.resourceInstance);
179 					break;
180 				}
181 		} else {
182 			final ConfigParamAtomic param = (ConfigParamAtomic) data;
183 			Configuration config = null;
184 			for (final Configuration cfg : this.resourceInstance
185 					.getConfigurations())
186 				if (cfg.getConfigParamAtomic() == param) {
187 					config = cfg;
188 					break;
189 				}
190 			if (config == null) {
191 				config = ModelManager.getInstance().createConfiguration(param);
192 				this.resourceInstance.addConfiguration(config);
193 			}
194 
195 			final String type = param.getType();
196 			if (type.equals("reference"))
197 				config.setValue(targetControler.getResourceInstance()
198 						.getCommonName());
199 			else if (type.equals("reference-array")) {
200 				final String[] entries = config.getValue().split(",");
201 				String added = "";
202 				String newValue = targetControler.getResourceInstance()
203 						.getCommonName();
204 				for (final String entrie : entries) {
205 					added += "," + entrie;
206 					if ((null != newValue) && newValue.equals(entrie))
207 						newValue = null;
208 				}
209 				if (newValue != null) {
210 					added += "," + newValue;
211 					config.setValue(added);
212 				}
213 			}
214 		}
215 	}

217 	public void onConnectionDeleted(final ConnectionEvent event) {
218 		final ResourceInstanceController targetControler = (ResourceInstanceController) event.target
219 				.getData();
220 		final Object data = event.sourceLabel.getData();
221 		if (data == null) {
222 			for (final VctController vctControler : this.views.keySet())
223 				if (this.views.get(vctControler) == event.source)
224 					targetControler.getResourceInstance().setParentInstance(
225 							null);
226 		} else {
227 			final ConfigParamAtomic param = (ConfigParamAtomic) data;
228 			final String type = param.getType();
229 			for (final Configuration config : this.resourceInstance
230 					.getConfigurations())
231 				if (config.getConfigParamAtomic() == param) {
232 					if (param.isReference())
233 						this.resourceInstance.removeConfiguration(config);
234 					else if (type.equals("reference-array")) {
235 						final String[] entries = config.getValue().split(",");
236 						String added = "";
237 						final String removeValue = targetControler
238 								.getResourceInstance().getCommonName();
239 						for (int i = 0; i < entries.length; i++)
240 							if (!removeValue.equals(entries[i]))
241 								added += "," + entries[i];
242 						if (!added.isEmpty())
243 							config.setValue(added);
244 						else
245 							this.resourceInstance.removeConfiguration(config);
246 					}
247 					break;
248 				}
249 		}
250 	}
