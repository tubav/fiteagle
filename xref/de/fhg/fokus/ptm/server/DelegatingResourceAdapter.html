33  import de.fhg.fokus.ptm.Identifier;
34  import de.fhg.fokus.ptm.PTMException;
35  
36  public abstract class DelegatingResourceAdapter extends AbstractResourceAdapter {
37  
38  	public abstract AbstractResource getResource(Identifier identifier) throws PTMException;
39  	
40  	public DelegatingResourceAdapter(Identifier parentId, Manager manager,
41  			String... types) throws PTMException {
42  		super(parentId, manager, types);
43  	}
44  
45  	@Override
46  	public void deleteResource(Identifier identifier) throws PTMException {
47  		getResource(identifier)._delete();
48  	}
49  
50  	@Override
51  	public Object getAttribute(Identifier identifier, String name)
52  			throws PTMException {
53  		return getResource(identifier)._getAttribute(name);
54  	}
55  
56  	@Override
57  	public void setAttribute(Identifier identifier, String name, Object value)
58  			throws PTMException {
59  		getResource(identifier)._setAttribute(name, value);
60  	}
61  
62  	@Override
63  	public Map<String, Object> getConfiguration(Identifier identifier)
64  			throws PTMException {
65  		return getResource(identifier)._getConfiguration();
66  	}
67  
68  	@Override
69  	public void setConfiguration(Identifier identifier,
70  			Map<String, Object> config) throws PTMException {
71  		getResource(identifier)._setConfiguration(config);
72  	}
73  }
