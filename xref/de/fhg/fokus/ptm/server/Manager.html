119 	public Resource addResource(Identifier parentId, String name, String type,
120 			Map<String, Object> config, Identifier owner) throws PTMException 
121 	{
122 		
123 		if (parentId == null)
124 			parentId = new Identifier(Identifier.DELIMITER);
125 		else if (parentId.isAdapter())
126 			throw new PTMException("Need a full id or null here, not: " + parentId);
127 		
128 		if (type == null || type.length() == 0)
129 			throw new PTMException("Illegal value for type: " + type);
130 
131 		if (config == null)
132 		{
133 			logger.warn("Config is null, using empty Map");
134 			config = new HashMap<String, Object>();
135 		}
136 		else if (name == null)
137 		{
138 			Object n = config.remove("name");
139 			if (n != null)
140 				name = n.toString();
141 		}
142 		
143 		Identifier aid = parentId.slash(type);
144 		
145 		ResourceAdapter a = getLocalAdapter(aid); 
146 		Resource r = a.addResource(parentId, name, type, config, owner);
147 		if (r == null)
148 			throw new PTMException("Failure adding resource. Adapter returned null.");
149 		return r;
150 	}

165 	@Override
166 	public Resource[] listResources(Identifier parentId, String type) throws PTMException
167 	{
168 		logger.debug("Listresources: " + parentId);
169 		Identifier aid;
170 		if (parentId != null)
171 		{
172 			assert_full(parentId);
173 			aid = parentId.slash();
174 		}
175 		else
176 			aid = parentId = new Identifier(Identifier.DELIMITER);
177 			
178 		if (type != null && type.length() == 0)
179 			type = null;
180 		
181 		
182 		ArrayList<Resource> resources = new ArrayList<Resource>();
183 		Map<Identifier, ResourceAdapter> adapters = getLocalAdapters(aid, type);
184 		
185 		for (Map.Entry<Identifier, ResourceAdapter> e : adapters.entrySet())
186 		{
187 			ResourceAdapter a = e.getValue();
188 			logger.debug("Listing: " + a + " - " + e.getKey());
189 			try
190 			{
191 				for (Resource r : a.listResources(parentId, type))
192 				{	
193 					logger.debug("adding resource: " + r);
194 					resources.add(r);
195 				}
196 			}
197 			catch (PTMException ex)
198 			{
199 				logger.error("Error while listing " + e.getKey(), ex);
200 			}
201 			logger.debug("complete list: " + resources);
202 		}
203 
204 		Resource[] result = resources.toArray(new Resource[0]);
205 		logger.debug("Retunring: " + result + " " + result.length);
206 		return result;
207 	}
